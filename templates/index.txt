    <script>
        let allSheep = [];
        let activeBreedFilter = '';
        let expandedCards = new Set();

        document.addEventListener('DOMContentLoaded', () => {
            fetchSheep();
            fetchBreeds();
        });

        function escapeHtml(str) {
            if (!str) return '';
            return String(str)
                .replace(/&/g, '&amp;')
                .replace(/</g, '&lt;')
                .replace(/>/g, '&gt;')
                .replace(/"/g, '&quot;')
                .replace(/'/g, '&#39;');
        }

        function showToast(message, type = 'success') {
            const container = document.getElementById('toastContainer');
            const toast = document.createElement('div');
            const colors = {
                success: 'bg-emerald-800 border-emerald-600 text-emerald-100',
                error:   'bg-red-900 border-red-700 text-red-100',
                info:    'bg-gray-700 border-gray-600 text-gray-100'
            };
            toast.className = `pointer-events-auto toast-enter px-4 py-3 rounded-lg border text-sm font-medium shadow-lg text-center w-full ${colors[type] || colors.info}`;
            toast.textContent = message;
            container.appendChild(toast);
            setTimeout(() => {
                toast.classList.remove('toast-enter');
                toast.classList.add('toast-exit');
                setTimeout(() => toast.remove(), 300);
            }, 3200);
        }

        function openFormPanel(reset = false) {
            if (reset) resetFormState();
            document.getElementById('formBackdrop').classList.remove('hidden');
            document.getElementById('formPanel').classList.remove('translate-x-full');
            requestAnimationFrame(() => {
                document.getElementById('formBackdrop').classList.remove('opacity-0');
            });
            document.body.style.overflow = 'hidden';
        }

        function closeFormPanel() {
            document.getElementById('formBackdrop').classList.add('opacity-0');
            document.getElementById('formPanel').classList.add('translate-x-full');
            document.body.style.overflow = '';
            setTimeout(() => document.getElementById('formBackdrop').classList.add('hidden'), 300);
        }

        async function fetchBreeds() {
            try {
                const response = await fetch('/api/breeds');
                const cloudBreeds = await response.json();
                const datalist = document.getElementById('breedOptions');
                const defaults = ['دروبر', 'سفلك', 'عساف', 'بلدي'];
                cloudBreeds.forEach(b => {
                    if (!defaults.includes(b)) {
                        const opt = document.createElement('option');
                        opt.value = b;
                        datalist.appendChild(opt);
                    }
                });
            } catch (err) { console.error(err); }
        }

        async function fetchSheep() {
            try {
                const response = await fetch('/api/sheep');
                allSheep = await response.json();
                updateStats();
                renderBreedFilters();
                applyFilters();
            } catch (error) {
                console.error(error);
                showToast('تعذر تحميل بيانات القطيع', 'error');
            }
        }

        function updateStats() {
            const breeds = new Set(allSheep.map(s => s.breed).filter(Boolean));
            const withNotes = allSheep.filter(s => s.notes && s.notes.trim()).length;
            document.getElementById('statTotal').textContent = allSheep.length;
            document.getElementById('statBreeds').textContent = breeds.size;
            document.getElementById('statNotes').textContent = withNotes;
        }

        function renderBreedFilters() {
            const container = document.getElementById('breedFilters');
            const breeds = [...new Set(allSheep.map(s => s.breed).filter(Boolean))].sort();
            const chipClass = (active) => active
                ? 'bg-blue-600 border-blue-500 text-white'
                : 'bg-gray-700 border-gray-600 text-gray-300 hover:bg-gray-600';

            let html = `<button data-breed="" class="breed-chip px-3 py-1 rounded-full text-xs font-medium border transition ${chipClass(!activeBreedFilter)}">الكل</button>`;
            breeds.forEach(b => {
                html += `<button data-breed="${escapeHtml(b)}" class="breed-chip px-3 py-1 rounded-full text-xs font-medium border transition ${chipClass(activeBreedFilter === b)}">${escapeHtml(b)}</button>`;
            });
            container.innerHTML = html;
        }

        function setBreedFilter(breed) {
            activeBreedFilter = breed;
            renderBreedFilters();
            applyFilters();
        }

        function applyFilters() {
            const query = document.getElementById('searchBar').value.trim().toLowerCase();
            let filtered = allSheep;

            if (activeBreedFilter) {
                filtered = filtered.filter(s => s.breed === activeBreedFilter);
            }
            if (query) {
                filtered = filtered.filter(s => {
                    const combined = `${s.tag_id} ${s.name} ${s.breed} ${s.birth_date} ${s.purchase_date} ${s.purchase_price} ${s.purchase_location} ${s.breeding_date} ${s.breeding_type} ${s.notes}`.toLowerCase();
                    return combined.includes(query);
                });
            }

            renderCards(filtered);
        }

        function hasPurchase(s) {
            return !!(s.purchase_date || s.purchase_price || s.purchase_location);
        }

        function hasBreeding(s) {
            return !!(s.breeding_date || s.breeding_type);
        }

        function hasNotes(s) {
            return !!(s.notes && s.notes.trim());
        }

        function renderCards(data) {
            const list = document.getElementById('sheepList');
            const empty = document.getElementById('emptyState');
            const resultsCount = document.getElementById('resultsCount');

            if (allSheep.length === 0) {
                list.innerHTML = '';
                empty.classList.remove('hidden');
                resultsCount.textContent = '';
                return;
            }

            empty.classList.add('hidden');

            const isFiltered = data.length !== allSheep.length || activeBreedFilter || document.getElementById('searchBar').value.trim();
            resultsCount.textContent = isFiltered ? `عرض ${data.length} من ${allSheep.length}` : `${allSheep.length} رأس`;

            if (data.length === 0) {
                list.innerHTML = `<div class="text-center py-12 text-gray-500 text-sm">لا توجد نتائج مطابقة للبحث أو الفلتر</div>`;
                return;
            }

            list.innerHTML = data.map(s => {
                const safeTag = escapeHtml(s.tag_id);
                const isExpanded = expandedCards.has(s.tag_id);
                const name = s.name ? escapeHtml(s.name) : '<span class="text-gray-500">بدون اسم</span>';
                const breed = s.breed
                    ? `<span class="bg-gray-900 border border-gray-600 px-2 py-0.5 rounded text-xs text-gray-300">${escapeHtml(s.breed)}</span>`
                    : '';

                const chips = [];
                if (hasPurchase(s)) chips.push('<span class="bg-emerald-900/40 border border-emerald-700/50 text-emerald-400 px-2 py-0.5 rounded text-xs">شراء</span>');
                if (hasBreeding(s)) chips.push('<span class="bg-purple-900/40 border border-purple-700/50 text-purple-400 px-2 py-0.5 rounded text-xs">تلقيح</span>');
                if (hasNotes(s)) chips.push('<span class="bg-amber-900/40 border border-amber-700/50 text-amber-400 px-2 py-0.5 rounded text-xs">ملاحظات</span>');
                const chipsHtml = chips.length ? `<div class="flex flex-wrap gap-1.5 mt-2">${chips.join('')}</div>` : '';

                const birthRow = s.birth_date
                    ? `<div class="flex justify-between text-sm"><span class="text-gray-500">تاريخ الولادة</span><span>${escapeHtml(s.birth_date)}</span></div>`
                    : '';

                const purchaseRows = hasPurchase(s) ? `
                    <div class="text-xs font-bold text-gray-500 mb-2 mt-1">بيانات الشراء</div>
                    ${s.purchase_date ? `<div class="flex justify-between text-sm mb-1"><span class="text-gray-500">التاريخ</span><span>${escapeHtml(s.purchase_date)}</span></div>` : ''}
                    ${s.purchase_price ? `<div class="flex justify-between text-sm mb-1"><span class="text-gray-500">السعر</span><span>${escapeHtml(s.purchase_price)}</span></div>` : ''}
                    ${s.purchase_location ? `<div class="flex justify-between text-sm mb-1"><span class="text-gray-500">المكان</span><span>${escapeHtml(s.purchase_location)}</span></div>` : ''}
                ` : `<div class="text-sm text-gray-500">لا توجد بيانات شراء</div>`;

                const breedingRows = hasBreeding(s) ? `
                    <div class="text-xs font-bold text-gray-500 mb-2 mt-1">بيانات التلقيح</div>
                    ${s.breeding_date ? `<div class="flex justify-between text-sm mb-1"><span class="text-gray-500">التاريخ</span><span>${escapeHtml(s.breeding_date)}</span></div>` : ''}
                    ${s.breeding_type ? `<div class="flex justify-between text-sm mb-1"><span class="text-gray-500">النوع</span><span>${escapeHtml(s.breeding_type)}</span></div>` : ''}
                ` : `<div class="text-sm text-gray-500">لا توجد بيانات تلقيح</div>`;

                const notesPreview = hasNotes(s)
                    ? `<p class="text-sm text-gray-400 line-clamp-2 leading-relaxed mb-2">${escapeHtml(s.notes.slice(0, 120))}${s.notes.length > 120 ? '…' : ''}</p>
                       <button data-action="notes" data-tag="${safeTag}" class="text-blue-400 hover:text-blue-300 text-xs font-semibold transition">عرض السجل الكامل ←</button>`
                    : `<span class="text-sm text-gray-500">لا توجد ملاحظات</span>`;

                return `
                <div class="bg-gray-900/60 border border-gray-700 rounded-xl overflow-hidden" data-tag="${safeTag}">
                    <div class="p-4 cursor-pointer select-none card-header">
                        <div class="flex items-start justify-between gap-3">
                            <div class="flex items-center gap-2 shrink-0">
                                <button data-action="edit" data-tag="${safeTag}" title="تعديل" class="w-8 h-8 flex items-center justify-center rounded-lg bg-yellow-600/20 hover:bg-yellow-600/40 text-yellow-400 transition text-sm">✏️</button>
                                <button data-action="delete" data-tag="${safeTag}" title="حذف" class="w-8 h-8 flex items-center justify-center rounded-lg bg-red-600/20 hover:bg-red-600/40 text-red-400 transition text-sm">🗑️</button>
                                <span class="text-gray-500 text-xs transition-transform duration-200 ${isExpanded ? 'rotate-180' : ''}">▼</span>
                            </div>
                            <div class="flex-1 text-right min-w-0">
                                <div class="flex items-center gap-2 flex-wrap justify-end">
                                    <span class="font-bold text-blue-400 text-base">${safeTag}</span>
                                    ${breed}
                                </div>
                                <div class="text-gray-300 text-sm mt-0.5">${name}</div>
                                ${chipsHtml}
                            </div>
                        </div>
                    </div>
                    <div class="card-details overflow-hidden border-t border-gray-700/50 ${isExpanded ? 'max-h-[600px] opacity-100' : 'max-h-0 opacity-0'}">
                        <div class="p-4 space-y-4 bg-gray-800/40 text-right">
                            ${birthRow}
                            <div class="grid grid-cols-1 sm:grid-cols-2 gap-4">
                                <div class="bg-gray-800 rounded-lg p-3 border border-gray-700/50">${purchaseRows}</div>
                                <div class="bg-gray-800 rounded-lg p-3 border border-gray-700/50">${breedingRows}</div>
                            </div>
                            <div class="bg-gray-800 rounded-lg p-3 border border-gray-700/50">
                                <div class="text-xs font-bold text-gray-500 mb-2">الملاحظات والسجل</div>
                                ${notesPreview}
                            </div>
                        </div>
                    </div>
                </div>`;
            }).join('');
        }

        function toggleCard(tagId) {
            if (expandedCards.has(tagId)) {
                expandedCards.delete(tagId);
            } else {
                expandedCards.add(tagId);
            }
            applyFilters();
        }

        function openNotesModal(tagId) {
            const sheep = allSheep.find(s => s.tag_id === tagId);
            if (!sheep || !sheep.notes) return;

            document.getElementById('modalHeader').innerText = `سجل وملاحظات الخروف رقم (${tagId})`;
            document.getElementById('modalBody').innerText = sheep.notes;

            const modal = document.getElementById('notesModal');
            modal.classList.remove('hidden');
            setTimeout(() => modal.classList.remove('opacity-0'), 10);
        }

        function closeNotesModal() {
            const modal = document.getElementById('notesModal');
            modal.classList.add('opacity-0');
            setTimeout(() => modal.classList.add('hidden'), 300);
        }

        function startEdit(tagId) {
            const sheep = allSheep.find(s => s.tag_id === tagId);
            if (!sheep) return;

            document.getElementById('tag_id').value = sheep.tag_id;
            document.getElementById('tag_id').disabled = true;
            document.getElementById('name').value = sheep.name || '';
            document.getElementById('breed').value = sheep.breed || '';
            document.getElementById('birth_date').value = sheep.birth_date || '';
            document.getElementById('purchase_date').value = sheep.purchase_date || '';
            document.getElementById('purchase_price').value = sheep.purchase_price || '';
            document.getElementById('purchase_location').value = sheep.purchase_location || '';
            document.getElementById('breeding_date').value = sheep.breeding_date || '';
            document.getElementById('breeding_type').value = sheep.breeding_type || '';
            document.getElementById('notes').value = sheep.notes || '';

            document.getElementById('formMode').value = 'UPDATE';
            document.getElementById('formTitle').innerText = `تعديل الخروف (${tagId})`;
            document.getElementById('formTitle').className = 'text-lg font-bold text-yellow-400';
            document.getElementById('submitBtn').innerText = 'تحديث البيانات';
            document.getElementById('submitBtn').className = 'w-full bg-yellow-600 hover:bg-yellow-700 transition font-bold py-3 px-4 rounded-lg text-white';
            document.getElementById('cancelBtn').classList.remove('hidden');

            openFormPanel();
        }

        function resetFormState() {
            document.getElementById('sheepForm').reset();
            document.getElementById('tag_id').disabled = false;
            document.getElementById('formMode').value = 'CREATE';
            document.getElementById('formTitle').innerText = 'تسجيل رأس جديد';
            document.getElementById('formTitle').className = 'text-lg font-bold text-blue-400';
            document.getElementById('submitBtn').innerText = 'حفظ البيانات';
            document.getElementById('submitBtn').className = 'w-full bg-blue-600 hover:bg-blue-700 transition font-bold py-3 px-4 rounded-lg';
            document.getElementById('cancelBtn').classList.add('hidden');
        }

        async function deleteSheep(tagId) {
            if (!confirm(`هل أنت متأكد من حذف الخروف رقم (${tagId}) نهائياً؟`)) return;

            try {
                const response = await fetch(`/api/sheep/${tagId}`, { method: 'DELETE' });
                const result = await response.json();
                showToast(result.message, result.success !== false ? 'success' : 'error');

                if (document.getElementById('tag_id').value === tagId && document.getElementById('formMode').value === 'UPDATE') {
                    closeFormPanel();
                    resetFormState();
                }
                expandedCards.delete(tagId);
                fetchSheep();
            } catch (err) {
                showToast('تعذر تنفيذ عملية الحذف', 'error');
            }
        }

        document.getElementById('breedFilters').addEventListener('click', (e) => {
            const chip = e.target.closest('[data-breed]');
            if (chip) setBreedFilter(chip.dataset.breed);
        });

        document.getElementById('sheepList').addEventListener('click', (e) => {
            const actionBtn = e.target.closest('[data-action]');
            if (actionBtn) {
                e.stopPropagation();
                const tag = actionBtn.dataset.tag;
                const action = actionBtn.dataset.action;
                if (action === 'edit') startEdit(tag);
                else if (action === 'delete') deleteSheep(tag);
                else if (action === 'notes') openNotesModal(tag);
                return;
            }
            const header = e.target.closest('.card-header');
            if (header) {
                const card = header.closest('[data-tag]');
                if (card) toggleCard(card.dataset.tag);
            }
        });

        document.getElementById('searchBar').addEventListener('input', applyFilters);

        document.getElementById('sheepForm').addEventListener('submit', async (e) => {
            e.preventDefault();
            const mode = document.getElementById('formMode').value;
            const tagId = document.getElementById('tag_id').value;

            const payload = {
                name: document.getElementById('name').value,
                breed: document.getElementById('breed').value,
                birth_date: document.getElementById('birth_date').value,
                purchase_date: document.getElementById('purchase_date').value,
                purchase_price: document.getElementById('purchase_price').value,
                purchase_location: document.getElementById('purchase_location').value,
                breeding_date: document.getElementById('breeding_date').value,
                breeding_type: document.getElementById('breeding_type').value,
                notes: document.getElementById('notes').value
            };

            let url = '/api/sheep';
            let method = 'POST';

            if (mode === 'UPDATE') {
                url = `/api/sheep/${tagId}`;
                method = 'PUT';
            } else {
                payload.tag_id = tagId;
            }

            try {
                const response = await fetch(url, {
                    method: method,
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify(payload)
                });
                const result = await response.json();

                if (result.success) {
                    showToast(result.message, 'success');
                    closeFormPanel();
                    resetFormState();
                    fetchSheep();
                    fetchBreeds();
                } else {
                    showToast(result.message, 'error');
                }
            } catch (error) {
                showToast('تعذر حفظ البيانات', 'error');
            }
        });
